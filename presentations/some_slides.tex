$Q_0$ the active quality factor\\
$T_a$ the amplifier noise temperature\\

\begin{columns}
\column{0.5\textwidth}
\centering
\begin{align*}
Q_0 = Q_L(1-\sqrt{G_l})^{-1} \\

\end{align*}
\column{0.5\textwidth}
\centering
\begin{align*}
T_{noise} = \frac{T_{cav} + G_l T_a}{1 + G_l - 2\sqrt{G_l}e^{-t/\tau - i\theta}}
\end{align*}
\end{columns}
